{{- $u := urls.Parse .Destination -}} {{- $src := $u.String -}} {{- $width := 0 -}} {{- $height := 0 -}} {{- $rasterFormats := slice "jpg" "jpeg" "png" "gif" "webp" "avif" "tif" "tiff" "bmp" -}} {{- if not $u.IsAbs -}} {{- $path := strings.TrimPrefix "./" $u.Path }} {{- with or (.PageInner.Resources.Get $path) (resources.Get $path) -}} {{- $src = .RelPermalink -}} {{- if in $rasterFormats .MediaType.SubType -}} {{- $width = .Width -}} {{- $height = .Height -}} {{- end -}} {{- with $u.RawQuery -}} {{- $src = printf "%s?%s" $src . -}} {{- end -}} {{- with $u.Fragment -}} {{- $src = printf "%s#%s" $src . -}} {{- end -}} {{- end -}} {{- end -}} {{- $attributes := merge .Attributes (dict "alt" .Text "src" $src "title" (.Title | transform.HTMLEscape) "loading" "lazy") -}} {{- if and $width $height -}} {{- $attributes = merge $attributes (dict "width" (printf "%d" $width) "height" (printf "%d" $height)) -}} {{- end -}} {{- /**/ -}}
